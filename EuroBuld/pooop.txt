  <script>
  async function loadServices() {
    try {
      const res = await fetch('/api/service');
      const services = await res.json();
      console.log(services);  // Логируем весь массив для проверки

      const container = document.getElementById('services-container');
      container.innerHTML = '';

      services.forEach(service => {
        const card = document.createElement('div');
        card.classList.add('product-card');

        // Фото
        if (service.ImageBase64) {
          const img = document.createElement('img');
          img.src = `data:image/jpg;base64,${service.ImageBase64}`; // добавляем префикс для правильного отображения
          card.appendChild(img);
        }

        // Название
        const name = document.createElement('div');
        name.classList.add('product-name');
        name.textContent = service.Item_Name;
        card.appendChild(name);

        // Цена
        const price = document.createElement('div');
        price.classList.add('product-price');
        price.textContent = `Цена: ${service.Price} ₽`;
        card.appendChild(price);

        // Кнопка "Подробнее"
        const btnDetails = document.createElement('button');
        btnDetails.classList.add('btn-details');
        btnDetails.textContent = 'Подробнее';
        btnDetails.onclick = function() {
          alert(`Подробнее о товаре: ${service.Item_Name}`);  // Здесь можно добавить логику отображения деталей товара
        };
        card.appendChild(btnDetails);

        container.appendChild(card);
      });
    } catch (err) {
      console.error('Ошибка при загрузке товаров:', err);
    }
  }

  window.onload = loadServices;
  </script>
